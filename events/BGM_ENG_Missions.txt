# Treaty of Troyes
country_event = {
	id = BGM_ENG_missions.1
	title = "BGM_ENG_missions.1.t"
	desc = "BGM_ENG_missions.1.d"
	picture = BAD_WITH_MONARCH_eventPicture

	major = yes

	is_triggered_only = yes
	
	option = {
		name = "BGM_ENG_missions.1.a" #Such a disaster
		if = {
			limit = { 
				exists = ENG
			}
			every_province = {
				limit = {
					OR = {
						owned_by = FRA
						owned_by_subject_of = FRA
					}
					OR = {
						area = normandy_area
						area = ile_de_france_area
						area = champagne_area
						area = picardy_area
					}
				}
				cede_province = ENG
			}
		}
		else_if = {
			limit = { exists = GBR }
			every_province = {
				limit = {
					OR = {
						owned_by = FRA
						owned_by_subject_of = FRA
					}
					OR = {
						area = normandy_area
						area = ile_de_france_area
						area = champagne_area
						area = picardy_area
					}
				}
				cede_province = GBR
			}
		}
	}
}

# Wycliffe
country_event = {
	id = BGM_ENG_missions.3
	title = "BGM_ENG_missions.3.t"
	desc = "BGM_ENG_missions.3.d"
	picture = BAD_WITH_MONARCH_eventPicture

	#major = yes

	trigger = {
		NOT = {
			has_global_flag = lollardy_started
		}
	}

	mean_time_to_happen = {
		years = 5
	}
	
	option = {
		name = "BGM_ENG_missions.3.a" #oh noes
		clr_global_flag = lollardy_started
		set_global_flag = lollardy_started
		custom_tooltip = lollardy_will_despawn

			limit = {
				is_capital = no
			}
			change_religion = lollard
			add_reform_center = lollard
			add_province_modifier = {
				name = conversion_zeal
				duration = 9125
			}
		}
		
	}
}
