	id = BGM_NAP_missions.1
	title = "BGM_NAP_missions.1.t"
	desc = "BGM_NAP_missions.1.d"
	picture = CIVIL_WAR_eventPicture

	trigger = {
		has_ruler_flag = giovanna_i
		OR = {
			NOT = { legitimacy = 25 }
			NOT = { stability = -2 }
		}
	}

		months = 9
	}
	
	option = {
		name = "BGM_NAP_missions.1.a"
		define_ruler = {
			dynasty = "d'Anjou"
			male = yes
			claim = 75
		}
		if = {
			limit = { 
				PRO = {
					is_subject_of_type = personal_union
					is_subject_of = ROOT
				}
			}
			MNE = {
				create_subject = {
					subject_type = personal_union
					who = PRO
				}
			}
		}
	}
}
